       y = "Flipper length (mm)") +
  theme_minimal()

ggplot2 extensions

patchwork

library(patchwork)

p1 <- ggplot(
  data = penguins, 
  aes(x = bill_length_mm, y = bill_depth_mm, color = species)
) +
  geom_point() +
  scale_color_manual(values = c("#19a6b3","#f26c0d","#5e3894")) +
  theme_minimal()

p2 <- ggplot(
  data = penguins, 
  aes(x = body_mass_g, fill = species)
) +
  geom_density(alpha = 0.5) +
  theme_minimal() +
  facet_grid(~ species)

p3 <- ggplot(
  data = penguins, 
  aes(x = species, y = body_mass_g, fill = species)
) +
  geom_boxplot() +
  theme_minimal()
